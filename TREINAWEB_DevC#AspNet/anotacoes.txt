Metodos com params //para receber argumentos indefinidos

	int Somar(params int[] numeros) //consjunto de dados recebidos, array numeros, o params deve ser sempre o último argumento
	{
		int resultado = 0;

		foreach (int numero in numeros)//varrer o array de numeros
		{
			resultado += numero; //somatorio de cada numero dentro do array numeros
		}

		return resultado;	
	}
